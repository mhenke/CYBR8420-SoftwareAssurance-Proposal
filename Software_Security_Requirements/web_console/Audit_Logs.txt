@startuml
actor Administrator
actor Victor

rectangle "Keycloak System" { 
    usecase "Manage Users" as ManageUsers
    usecase "Unauthorized User Modification" as UnauthorizedModification
    usecase "Audit Logs" as AuditLogs
}

Administrator --> AuditLogs : Reviews Audit Logs
Victor --> UnauthorizedModification : Maliciously Modify User Roles

UnauthorizedModification -[dashed]-> ManageUsers : <<threatens>>
ManageUsers -[dashed]-> AuditLogs : <<mitigates>>
@enduml
